149 private void RemoveConflictingJobs(AudioType type) {
150 // cancel the job if one exists with the same type
151 if (_jobTable.ContainsKey(type)) {

176 private IEnumerator RunAudioJob(AudioJob job) {
177 if (job.delay != null) yield return job.delay;
178
179 AudioTrack _track = GetAudioTrack(job.type); // track existence should be verified by now
180 _track.source.clip = GetAudioClipFromAudioTrack(job.type, _track);
181
182 float _initial = 0f;
183 float _target = 1f;
184 switch (job.action) {
185 case AudioAction.START:
186 _track.source.Play();
187 break;
188 case AudioAction.STOP when !job.fade:
189 _track.source.Stop();
190 break;
191 case AudioAction.STOP:
192 _initial = 1f;
193 _target = 0f;
194 break;
195 case AudioAction.RESTART:
196 _track.source.Stop();
197 _track.source.Play();
198 break;
199 }
200
201 // fade volume
202 if (job.fade) {
203 float _duration = 1.0f;
204 float _timer = 0.0f;
205
206 while (_timer <= _duration) {
207 _track.source.volume = Mathf.Lerp(_initial, _target, _timer / _duration);
208 _timer += Time.deltaTime;
209 yield return null;
210 }
211
212 // if _timer was 0.9999 and Time.deltaTime was 0.01 we would not have reached the target
213 // make sure the volume is set to the value we want
214 _track.source.volume = _target;
215
216 if (job.action == AudioAction.STOP) {
217 _track.source.Stop();
218 }
219 }
220
221 _jobTable.Remove(job.type);
222 Log("Job count: "+_jobTable.Count);
223 }
