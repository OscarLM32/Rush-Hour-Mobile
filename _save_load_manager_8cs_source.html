219 private void Load()
220 {
221 if (File.Exists(_savePath))
222 {
223 FileStream stream = new FileStream(_savePath, FileMode.Open);
224
225 if (stream.Length == 0)
226 {
227 AppLogger.LogError(DebugFlag.SAVE_LOAD_SYSTEM, "The stream you are trying to read is empty");
228 return;
229 }
230
231 _gameData = _formatter.Deserialize(stream) as GameData;
232 stream.Close();
233
234 _tempGameData = new TemporalGameData();
235 if (!_gameData.levelClassifiedDocumentsBitmap.ContainsKey(currentSceneName)) return;
236 _tempGameData.classifiedDocumentsBitmap = _gameData.levelClassifiedDocumentsBitmap[currentSceneName];
237 }
238 else
239 {
240 _gameData = new GameData();
241 _tempGameData = new TemporalGameData();
242 Debug.Log("The save file was not found in path: " + _savePath);
243 }
244 }

273 private int CurrentLevelToLevelInt()
274 {
275 //All the "level" scenes are named "Level" followed by the number with no space.
276 string sNum = currentSceneName.Substring(5); //It can be also written as [5..]
277 int lvl = 0;
278 try
279 {
280 lvl = int.Parse(sNum);

282 catch (Exception e)

284 AppLogger.Log(DebugFlag.SAVE_LOAD_SYSTEM, e.ToString());
