117 public void CreateEmailPasswordAccount(string email, string password, Action OnSuccess, Action OnFailure)
118 {
119 auth.CreateUserWithEmailAndPasswordAsync(email, password).ContinueWith(task => {
120 if (task.IsCanceled) {
121 Debug.LogError("CreateUserWithEmailAndPasswordAsync was canceled.");
122 _taskResult = false;
123 OnFailure();
124 return;
125 }
126 if (task.IsFaulted) {
127 Debug.LogError("CreateUserWithEmailAndPasswordAsync encountered an error: " + task.Exception);
128 _taskResult = false;
129 OnFailure();
130 return;
131 }
132
133 // Firebase user has been created.
134 Firebase.Auth.AuthResult result = task.Result;
135 Debug.LogFormat("Firebase user created successfully: {0} ({1})",
136 result.User.DisplayName, result.User.UserId);
137
138 OnSuccess();
139
140 User.Instance.SetUpUser(result.User);
141 db.Child(DatabaseFields.USER.ToString()).Child(User.Instance.uid).SetRawJsonValueAsync(User.Instance.GetJsonUserData());
142 });
143 }
